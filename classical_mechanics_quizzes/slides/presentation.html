from manim import *

class KineticScene(Scene):
    def construct(self):
        title = Text("Kinetic Energy", font_size=48, color=BLUE).to_edge(UP)
        eq = MathTex("KE = \tfrac{1}{2} m v^2", font_size=54).next_to(title, DOWN)
        self.play(FadeIn(title), FadeIn(eq))
        ball = Circle(radius=0.3, color=RED).shift(LEFT*3)
        arrow = Arrow(ball.get_center(), ball.get_center()+RIGHT*2)
        label = Text("KE increases with speed", font_size=28).next_to(arrow, UP)
        self.play(FadeIn(ball))
        self.play(GrowArrow(arrow), Write(label))
        self.play(ball.animate.shift(RIGHT*4), run_time=3)
        self.wait(0.5)

class PotentialScene(Scene):
    def construct(self):
        title = Text("Potential Energy", font_size=48, color=GREEN).to_edge(UP)
        eq = MathTex("PE = m g h", font_size=54).next_to(title, DOWN)
        self.play(FadeIn(title), FadeIn(eq))
        hill = Polygon(LEFT*3+DOWN, ORIGIN+UP*2, RIGHT*3+DOWN, color=GREEN)
        ball = Circle(radius=0.3, color=RED).shift(LEFT*2+UP*0.7)
        label = Text("PE increases with height", font_size=28).next_to(ball, UP)
        self.play(FadeIn(hill), FadeIn(ball), FadeIn(label))
        self.play(ball.animate.shift(RIGHT*4+DOWN*2), run_time=3)
        self.wait(0.5)

class SpringScene(Scene):
    def construct(self):
        title = Text("Elastic Potential", font_size=44).to_edge(UP)
        spring = Rectangle(width=0.25, height=2, color=GRAY)
        label = Text("Elastic Potential Energy", font_size=28).next_to(spring, RIGHT)
        self.play(FadeIn(title))
        self.play(FadeIn(spring), FadeIn(label))
        self.wait(1)
